import React from 'react';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import TasksScreen from '../screens/TasksScreen';
import EnergyMappingScreen from '../screens/EnergyMappingScreen';
import HyperfocusScreen from '../screens/HyperfocusScreen';

const Tab = createBottomTabNavigator();

export default function AppStack() {
  return (
    <Tab.Navigator>
      <Tab.Screen name="Tasks" component={TasksScreen} />
      <Tab.Screen name="Energy" component={EnergyMappingScreen} />
      <Tab.Screen name="Hyperfocus" component={HyperfocusScreen} />
    </Tab.Navigator>
  );
}
